```cpp
#include <iostream>

using namespace std;

const int N = 1e6 + 10;

int n;
int q[N];

void quick_sort(int q[], int l, int r) {
    if (l >= r) return;
    
    int x = q[(l + r) / 2], i = l - 1, j = r + 1;
    while (i < j) { // 只要i,j没有相遇
        do i++; while (q[i] < x); // 如果q[i] < x，i++
        do j--; while (q[j] > x); // 如果q[j] > x，j--
        if (i < j) swap(q[i], q[j]); // 交换
    }
    
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) scanf("%d", &q[i]);
    
    quick_sort(q, 0, n - 1);
    
    for (int i = 0; i < n; ++i) printf("%d ", q[i]);
    
    return 0;
}
```

```cpp
#include <iostream>

using namespace std;

const int N = 1e6 + 10;

int n;
int q[N], tmp[N];

void merge_sort(int q[], int l, int r) {
    if (l >= r) return;
    
    int mid = l + r >> 1;
    merge_sort(q, l, mid), merge_sort(q, mid + 1, r); // 先递归
    
    int k = 0, i = l, j = mid + 1; // k已经合并了几个数,i，j两个起点
    while (i <= mid && j <= r) {
        if (q[i] < q[j]) tmp[k++] = q[i++];
        else tmp[k++] = q[j++];
    }
    while (i <= mid) tmp[k++] = q[i++];
    while (j <= r) tmp[k++] = q[j++];
    
    for (int i = l, j = 0; i <= r; ++i, ++j) q[i] = tmp[j];
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) scanf("%d", &q[i]);
    
    merge_sort(q, 0, n - 1);
    
    for (int i = 0; i < n; ++i) printf("%d ", q[i]);
    
    return 0;
}
```
